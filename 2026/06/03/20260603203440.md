+                    sb.append('"')
+                    i += 2
+                }
+
+                c == '"' -> {
+                    inQuotes = !inQuotes
+                    i++
+                }
+
+                c == ',' && !inQuotes -> {
+                    fields.add(sb.toString())
+                    sb.clear()
+                    i++
+                }
+
+                else -> {
+                    sb.append(c)
+                    i++
+                }
+            }
+        }
+        fields.add(sb.toString())
+        return fields
+    }
