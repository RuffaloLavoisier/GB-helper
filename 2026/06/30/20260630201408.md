+    @Test

+    @Test
+    public void accumulatesPositiveActiveCaloriesFromDayStart() {

+                sample(START_TS + 240, 1800)
+        );
+
+        final List<Entry> entries = CaloriesDailyFragment.createCumulativeActiveCaloriesEntries(
+                samples,
+                tsTranslation,
+                START_TS
+        );

+        assertEquals(5, entries.size());
